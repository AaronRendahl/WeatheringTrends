A basic example

library(WeatheringTrends)
data(welldata)
fit1 <- FitElementRatio("Sr", "Zr", "depth.top", welldata)
## fitting Sr/Zr
fit1$par
##         p1         x2          c        sd1        sd2          d 
##  0.1645912  6.9942994  1.2841246  0.2448479  0.3158730 -0.3298341
fit1$output
##     depth1     depth2          c  logratio1  logratio2        sd1 
##  1.1512004  6.9942994  1.2841246 -1.6139587 -0.3298341  0.2448479 
##        sd2 
##  0.3158730
plot(fit1)

fit1 <- profile(fit1, "x2")
fit1$confint
##       lower    upper
## x2 4.785353 10.01146
plot(fit1)

Can fit multiple fits at once

fits <- FitElementRatios(c("Sr", "Pb"), c("Zr", "V"), "depth.top", welldata)
## fitting Sr/Zr
## fitting Sr/V
## fitting Pb/Zr
## fitting Pb/V
par(mfrow=c(2,2))
for(i in 1:2) for(j in 1:2) plot(fits[[i]][[j]])
