A basic example

library(WeatheringTrends)
data(welldata)
fit1 <- FitElementRatio("Sr", "Zr", "depth.top", welldata)
## fitting Sr/Zr
fit1$par
##          p          d          c         s1         s2          r 
##  0.1645912  6.9942994  1.2841246  0.2448479  0.3158730 -0.3298341
fit1$output
##     depth1     depth2  logratio1  logratio2         s1         s2 
##  1.1512004  6.9942994 -1.6139587 -0.3298341  0.2448479  0.3158730
fit1$confint
##      lower    upper
## d 4.785353 10.01146
plot(fit1)

Can fit multiple fits at once, turn off profiling of depth for this

fits <- FitElementRatios(c("Sr", "Pb"), c("Zr", "V"), "depth.top", welldata, profile=FALSE)
## fitting Sr/Zr
## fitting Sr/V
## fitting Pb/Zr
## fitting Pb/V
par(mfrow=c(2,2))
for(i in 1:2) for(j in 1:2) plot(fits[[i]][[j]])
